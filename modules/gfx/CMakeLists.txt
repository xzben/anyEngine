set(TARGET_NAME gfx)

set(GFX_LIB_SOURCE "${CMAKE_CURRENT_SOURCE_DIR}")

file(GLOB_RECURSE RUNTIME_HEADERS CONFIGURE_DEPENDS ${GFX_LIB_SOURCE}/*.h)
file(GLOB_RECURSE RUNTIME_SOURCES CONFIGURE_DEPENDS ${GFX_LIB_SOURCE}/*.cpp)
file(GLOB_RECURSE RUNTIME_GLAB   ${BYGFX_THIRD_LIB_DIR}/glad/src/glad.c)
file(GLOB_RECURSE RUNTIME_GLAB_HEADS CONFIGURE_DEPENDS ${BYGFX_THIRD_LIB_DIR}/glad/include/*.h)

set(GFX_PRIVATE_INCLUDE 
${GFX_LIB_SOURCE}
${THIRD_LIB_INCLUDE}
${THIRD_LIB_INCLUDE}/glad/include)

source_group(TREE "${GFX_LIB_SOURCE}" FILES 
	${RUNTIME_HEADERS} 
	${RUNTIME_SOURCES})

source_group(TREE "${BYGFX_THIRD_LIB_DIR}" FILES 
	${RUNTIME_GLAB} 
	${RUNTIME_GLAB_HEADS})

add_library(${TARGET_NAME} ${RUNTIME_HEADERS} 
	${RUNTIME_SOURCES}
    ${RUNTIME_GLAB})

target_include_directories(${TARGET_NAME} PUBLIC ${GFX_LIB_SOURCE})
target_include_directories(${TARGET_NAME} PRIVATE ${GFX_PRIVATE_INCLUDE})

target_link_libraries(${TARGET_NAME} PUBLIC glfw)
